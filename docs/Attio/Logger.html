def default_formatter
  proc do |severity, datetime, progname, msg|
    data = {
      timestamp: datetime.iso8601,
      level: severity,
      progname: progname
    }

    if msg.is_a?(Hash)
      data.merge!(msg)
    else
      data[:message] = msg
    end

    "#{JSON.generate(data)}\n"
  end
end

def format_message(message, context)
  return message if context.empty?
  
  {
    message: message,
    **context
  }
end
